import React, { useState, useEffect } from 'react';
import { Link } from 'react-router-dom';
import { createPageUrl } from '@/utils';
import { Project } from '@/entities/Project';
import { Card, CardHeader, CardTitle, CardContent, CardDescription } from '@/components/ui/card';
import { Button } from '@/components/ui/button';
import { PlusCircle, Code, Music, Shield, Zap, GitBranch, Mic } from 'lucide-react';

const QuickAccessTile = ({ title, icon, description, url }) => (
  <Link to={url} className="block">
    <Card className="bg-slate-900/50 backdrop-blur-xl border-slate-800 h-full hover:border-indigo-500/50 transition-colors duration-300">
      <CardHeader>
        <div className="flex items-center gap-3">
          {React.createElement(icon, { className: "w-6 h-6 text-indigo-400" })}
          <CardTitle className="text-lg text-white">{title}</CardTitle>
        </div>
      </CardHeader>
      <CardContent>
        <p className="text-slate-400 text-sm">{description}</p>
      </CardContent>
    </Card>
  </Link>
);

export default function Dashboard() {
  const [projects, setProjects] = useState([]);
  const [isLoading, setIsLoading] = useState(true);

  useEffect(() => {
    const fetchProjects = async () => {
      setIsLoading(true);
      try {
        const recentProjects = await Project.list('-created_date', 5);
        setProjects(recentProjects);
      } catch (error) {
        console.error("Failed to fetch projects:", error);
      }
      setIsLoading(false);
    };
    fetchProjects();
  }, []);

  return (
    <div className="p-6 space-y-8">
      {/* Header */}
      <div className="flex items-center justify-between">
        <div>
          <h1 className="text-3xl font-bold text-white">Dashboard</h1>
          <p className="text-slate-400">Welcome back! Here's a snapshot of your workspace.</p>
        </div>
        <Button className="bg-indigo-600 hover:bg-indigo-700">
          <PlusCircle className="w-4 h-4 mr-2" />
          New Project
        </Button>
      </div>

      {/* Quick Access */}
      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-6">
        <QuickAccessTile 
          title="Music Studio"
          icon={Music}
          description="Compose melodies and arrange your next hit."
          url={createPageUrl("Studio")}
        />
        <QuickAccessTile 
          title="Code Translator"
          icon={Code}
          description="Convert code between 14+ languages instantly."
          url={createPageUrl("CodeTranslator")}
        />
        <QuickAccessTile 
          title="AI Assistant"
          icon={Zap}
          description="Chat with an AI to spark creative ideas."
          url={createPageUrl("AIAssistant")}
        />
        <QuickAccessTile 
          title="Security Scanner"
          icon={Shield}
          description="Analyze your code for vulnerabilities."
          url={createPageUrl("SecurityScanner")}
        />
      </div>

      {/* Recent Projects & Analytics */}
      <div className="grid lg:grid-cols-3 gap-6">
        <Card className="lg:col-span-2 bg-slate-900/50 backdrop-blur-xl border-slate-800">
          <CardHeader>
            <CardTitle className="text-white">Recent Projects</CardTitle>
          </CardHeader>
          <CardContent>
            {isLoading ? (
              <p className="text-slate-400">Loading projects...</p>
            ) : projects.length > 0 ? (
              <ul className="space-y-4">
                {projects.map(p => (
                  <li key={p.id} className="flex items-center justify-between p-3 bg-slate-800/50 rounded-lg">
                    <div>
                      <h4 className="font-semibold text-white">{p.title}</h4>
                      <p className="text-sm text-slate-400">{p.description}</p>
                    </div>
                    <div className="flex items-center gap-2">
                      <span className="text-xs text-slate-500">BPM: {p.bpm}</span>
                      <Button variant="ghost" size="sm">Open</Button>
                    </div>
                  </li>
                ))}
              </ul>
            ) : (
              <p className="text-slate-400">You haven't created any projects yet.</p>
            )}
          </CardContent>
        </Card>

        <Card className="bg-slate-900/50 backdrop-blur-xl border-slate-800">
          <CardHeader>
            <CardTitle className="text-white">Analytics</CardTitle>
            <CardDescription className="text-slate-500">Your activity this week</CardDescription>
          </CardHeader>
          <CardContent className="space-y-4">
            <div className="flex justify-between items-center">
              <span className="text-slate-300">Code Translated</span>
              <span className="font-bold text-white">1,204 lines</span>
            </div>
            <div className="flex justify-between items-center">
              <span className="text-slate-300">Tracks Composed</span>
              <span className="font-bold text-white">3</span>
            </div>
            <div className="flex justify-between items-center">
              <span className="text-slate-300">Vulnerabilities Found</span>
              <span className="font-bold text-yellow-400">8</span>
            </div>
             <div className="flex justify-between items-center">
              <span className="text-slate-300">AI Queries</span>
              <span className="font-bold text-white">72</span>
            </div>
          </CardContent>
        </Card>
      </div>
    </div>
  );
}
